# 抢课助手配置文件
# 请根据需要修改以下配置

# 登录信息
username=
password=

# 选课设置
# 课程类型：1=专业课，2=公选课，留空则手动选择
course_type=2
# 是否过滤已满课程：y=是，n=否，留空则手动选择
enable_filter=y
# 课程名称：留空则手动输入
course_name=

# 课程屏蔽设置
# 屏蔽课程列表：用逗号分隔多个课程名称，留空则不屏蔽任何课程
# 示例：blocked_courses=高等数学,大学物理,线性代数
# 当搜索结果中出现这些课程时，程序会自动跳过不选择
blocked_courses=

# 延时设置（单位：秒）
# 搜索间隔时间
search_interval=2
# 点击间隔时间
click_interval=0.5
# 过滤器等待时间
filter_wait=0.3
# 页面加载等待时间
page_load_wait=2

# 重试设置
# 浏览器重试次数（设置为999时无限重试）
browser_retries=3
# 登录重试次数（设置为999时无限重试）
login_retries=3
# 选课按钮查找最大尝试次数（设置为999时无限尝试）
enter_course_attempts=10

# 说明：
# 1. username和password为空时，程序会提示手动输入
# 2. course_type为空时，程序会提示选择课程类型（1=专业课，2=公选课）
# 3. enable_filter为空时，程序会提示是否过滤已满课程
# 4. course_name为空时，程序会提示手动输入课程名称
# 5. blocked_courses为空时，不屏蔽任何课程；设置后会自动跳过包含这些关键词的课程
# 6. 延时时间可以根据网络情况调整，网络慢可以适当增加
# 7. 重试次数可以根据需要调整，设置为999时表示无限重试直到用户手动终止（Ctrl+C）
# 8. 无限重试模式适用于网络不稳定或需要长时间等待的情况
# 9. 使用无限重试时请确保有足够的耐心，可随时按Ctrl+C终止
# 10. 课程屏蔽功能支持模糊匹配，只要课程名称包含屏蔽列表中的关键词就会被跳过
